-- =========================
-- GENERIC ID GENERATOR
-- =========================
CREATE OR REPLACE FUNCTION generate_prefixed_id(
  seq_name TEXT,
  prefix TEXT,
  pad_length INTEGER DEFAULT 4
)
RETURNS VARCHAR(20)
LANGUAGE plpgsql
AS $$
DECLARE
  next_num BIGINT;
BEGIN
  EXECUTE FORMAT('SELECT nextval(%L)', seq_name) INTO next_num;
  RETURN prefix || LPAD(next_num::TEXT, pad_length, '0');
END;
$$;

-- =========================
-- USERS
-- =========================
CREATE TABLE users (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('users_id_seq','U',4),
  email VARCHAR(255) UNIQUE NOT NULL,
  name VARCHAR(255) NOT NULL,
  avatar_url TEXT,
  role VARCHAR(20) NOT NULL DEFAULT 'user'
    CHECK (role IN ('user','seller','admin')),
  is_active BOOLEAN NOT NULL DEFAULT TRUE,
  is_online BOOLEAN NOT NULL DEFAULT FALSE,
  phone_number VARCHAR(20),
  address TEXT,
  city VARCHAR(100),
  postal_code VARCHAR(20),
  country VARCHAR(100),
  bio TEXT,
  last_seen_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- CATEGORIES
-- =========================
CREATE TABLE categories (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('categories_id_seq','CAT',4),
  name VARCHAR(100) UNIQUE NOT NULL,
  description TEXT,
  is_enabled BOOLEAN NOT NULL DEFAULT TRUE,
  sort_priority INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- SUBCATEGORIES
-- =========================
CREATE TABLE subcategories (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('subcategories_id_seq','SCAT',4),
  category_id VARCHAR(20) NOT NULL REFERENCES categories(id) ON DELETE CASCADE,
  name VARCHAR(100) NOT NULL,
  description TEXT,
  is_enabled BOOLEAN NOT NULL DEFAULT TRUE,
  sort_priority INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(category_id,name)
);


-- =========================
-- PRODUCTS (BASE)
-- =========================
CREATE TABLE products (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('products_id_seq','P',4),
  title VARCHAR(255) NOT NULL,
  description TEXT NOT NULL,
  base_price NUMERIC(10,2) NOT NULL CHECK (base_price > 0),
  total_stock INT NOT NULL DEFAULT 0 CHECK (total_stock >= 0),

  category_id VARCHAR(20) REFERENCES categories(id) ON DELETE SET NULL,
  subcategory_id VARCHAR(20) REFERENCES subcategories(id) ON DELETE SET NULL,
  seller_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,

  condition VARCHAR(20) NOT NULL
    CHECK (condition IN ('like_new','excellent','good','fair')),

  image_urls TEXT[],

  trade_preference TEXT[] NOT NULL DEFAULT ARRAY['face_to_face']
    CHECK (
      trade_preference <@ ARRAY['face_to_face','delivery_official','delivery_self']
      AND array_length(trade_preference,1) > 0
    ),

  open_to_offers BOOLEAN NOT NULL DEFAULT FALSE,

  status VARCHAR(20) NOT NULL DEFAULT 'active'
    CHECK (status IN ('draft','active','reserved','sold','inactive')),

  view_count INT NOT NULL DEFAULT 0 CHECK (view_count >= 0),
  likes_count INT NOT NULL DEFAULT 0 CHECK (likes_count >= 0),

  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- PRODUCT VARIANTS (OPTION A)
-- =========================
CREATE TABLE product_variants (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('product_variants_id_seq','PV',4),
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  sku VARCHAR(100) UNIQUE,
  price NUMERIC(10,2),
  quantity INT NOT NULL DEFAULT 0 CHECK (quantity >= 0),
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- VARIANT ATTRIBUTES
-- =========================
CREATE TABLE product_variant_attributes (
  id BIGSERIAL PRIMARY KEY,
  variant_id VARCHAR(20) NOT NULL REFERENCES product_variants(id) ON DELETE CASCADE,
  attribute_name VARCHAR(50) NOT NULL,
  attribute_value VARCHAR(100) NOT NULL,
  UNIQUE(variant_id, attribute_name)
);


-- =========================
-- PRODUCT MEETUP LOCATIONS
-- =========================
CREATE TABLE product_meetup_locations (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('product_meetup_locations_id_seq','ML',4),
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  location_name VARCHAR(255) NOT NULL,
  address TEXT,
  city VARCHAR(100),
  state VARCHAR(100),
  postal_code VARCHAR(20),
  latitude NUMERIC(10,7),
  longitude NUMERIC(10,7),
  is_default BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- CART ITEMS
-- =========================
CREATE TABLE cart_items (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('cart_items_id_seq','CI',4),
  user_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  variant_id VARCHAR(20) REFERENCES product_variants(id) ON DELETE SET NULL,
  quantity INT NOT NULL DEFAULT 1 CHECK (quantity > 0),
  added_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, product_id, variant_id)
);


-- =========================
-- ORDERS
-- =========================
CREATE TABLE orders (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('orders_id_seq','O',4),
  order_number VARCHAR(50) UNIQUE NOT NULL,
  buyer_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  total_price NUMERIC(10,2) NOT NULL DEFAULT 0 CHECK (total_price >= 0),
  status VARCHAR(30) NOT NULL DEFAULT 'pending'
    CHECK (status IN ('pending','paid','pending_handover','completed','cancelled','disputed')),
  payment_status VARCHAR(20) NOT NULL DEFAULT 'pending'
    CHECK (payment_status IN ('pending','paid','failed','refunded')),
  handover_location TEXT,
  handover_date TIMESTAMPTZ,
  notes TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- ORDER ITEMS
-- =========================
CREATE TABLE order_items (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('order_items_id_seq','OI',4),
  order_id VARCHAR(20) NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE RESTRICT,
  variant_id VARCHAR(20) REFERENCES product_variants(id) ON DELETE RESTRICT,
  quantity INT NOT NULL DEFAULT 1 CHECK (quantity > 0),
  unit_price NUMERIC(10,2) NOT NULL CHECK (unit_price > 0),
  subtotal NUMERIC(10,2) NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- PAYMENTS
-- =========================
CREATE TABLE payments (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('payments_id_seq','PAY',4),
  order_id VARCHAR(20) NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  user_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  amount NUMERIC(10,2) NOT NULL CHECK (amount > 0),
  payment_method VARCHAR(30)
    CHECK (payment_method IS NULL OR payment_method IN ('card','apple_google_pay','campus_wallet','bank_transfer','cash')),
  payment_status VARCHAR(20) NOT NULL DEFAULT 'pending'
    CHECK (payment_status IN ('pending','paid','failed','refunded')),
  transaction_id VARCHAR(255) UNIQUE,
  gateway_response JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- CHAT
-- =========================
CREATE TABLE chat_conversations (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('chat_conversations_id_seq','CC',4),
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  buyer_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  seller_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  deal_agreed BOOLEAN NOT NULL DEFAULT FALSE,
  last_message_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(product_id,buyer_id,seller_id),
  CHECK (buyer_id <> seller_id)
);


CREATE TABLE chat_messages (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('chat_messages_id_seq','CM',4),
  conversation_id VARCHAR(20) NOT NULL REFERENCES chat_conversations(id) ON DELETE CASCADE,
  sender_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  message_text TEXT,
  is_image BOOLEAN NOT NULL DEFAULT FALSE,
  image_url TEXT,
  is_read BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CHECK (
    (message_text IS NOT NULL AND message_text <> '') OR image_url IS NOT NULL
  ),
  CHECK (
    (is_image = TRUE AND image_url IS NOT NULL) OR (is_image = FALSE)
  )
);


-- =========================
-- NOTIFICATIONS
-- =========================
CREATE TABLE notifications (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('notifications_id_seq','N',4),
  user_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  title VARCHAR(255) NOT NULL,
  message TEXT NOT NULL,
  notification_type VARCHAR(30)
    CHECK (notification_type IS NULL OR notification_type IN ('order','message','system','item_sold','item_bought')),
  related_order_id VARCHAR(20) REFERENCES orders(id) ON DELETE SET NULL,
  related_product_id VARCHAR(20) REFERENCES products(id) ON DELETE SET NULL,
  related_conversation_id VARCHAR(20) REFERENCES chat_conversations(id) ON DELETE SET NULL,
  is_read BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- REVIEWS
-- =========================
CREATE TABLE reviews (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('reviews_id_seq','R',4),
  order_id VARCHAR(20) NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  reviewer_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  reviewee_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  rating INT NOT NULL CHECK (rating BETWEEN 1 AND 5),
  title VARCHAR(255),
  comment TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(order_id, reviewer_id),
  CHECK (reviewer_id <> reviewee_id)
);

  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('seller_profiles_id_seq','SP',4),
  user_id VARCHAR(20) NOT NULL UNIQUE REFERENCES users(id) ON DELETE CASCADE,
  total_sales INT NOT NULL DEFAULT 0,
  average_rating NUMERIC(3,2) NOT NULL DEFAULT 0,
  total_reviews INT NOT NULL DEFAULT 0,
  total_products INT NOT NULL DEFAULT 0,
  is_verified BOOLEAN NOT NULL DEFAULT FALSE,
  verification_date TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


-- =========================
-- FAVORITES
-- =========================
CREATE TABLE favorites (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('favorites_id_seq','F',4),
  user_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  product_id VARCHAR(20) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, product_id)
);


-- =========================
-- DISPUTES
-- =========================
CREATE TABLE disputes (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('disputes_id_seq','D',4),
  order_id VARCHAR(20) NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  reporter_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  accused_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  reason VARCHAR(255) NOT NULL,
  description TEXT NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'open'
    CHECK (status IN ('open','in_review','resolved','closed')),
  admin_notes TEXT,
  resolution_notes TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  resolved_at TIMESTAMPTZ,
  CHECK (reporter_id <> accused_id)
);


-- =========================
-- ADMIN LOGS
-- =========================
CREATE TABLE admin_logs (
  id VARCHAR(20) PRIMARY KEY DEFAULT generate_prefixed_id('admin_logs_id_seq','AL',4),
  admin_id VARCHAR(20) NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
  action VARCHAR(100) NOT NULL,
  entity_type VARCHAR(20)
    CHECK (entity_type IS NULL OR entity_type IN ('user','product','order','dispute')),
  entity_id VARCHAR(20),
  details JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);
